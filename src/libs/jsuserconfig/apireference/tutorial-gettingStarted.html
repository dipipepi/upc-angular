Tutorial: Getting started

Call stack

   //Provide Client's config usually obtained from Resource Discovery
   var clientConfig = new AvayaUserClient.Config.ClientConfig(resources);
   
   var client = new AvayaUserClient(clientConfig);
   var service = client.userService;
   
   service.changePassword(null, '3C26E5B3273911E68239DD9AA57729EF', '321321').fail(function (response) {
       console.log('Guest changePassword: %o:', response);
   }).done(function (response) {
       console.log('Guest changePassword: %o:', response);
   });
   
   service.start();
